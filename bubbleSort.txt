#include <stdio.h>
#include<stdlib.h> 
#include<time.h>
#include<unistd.h>
void swap(int *xp, int *yp)
{
    int temp = *xp;
    *xp = *yp;
    *yp = temp;
}
 

void bubbleSort(int arr[], int n)
{
   int i, j;
   for (i = 0; i < n-1; i++)     
       for (j = 0; j < n-i-1; j++)
           if (arr[j] > arr[j+1])
              swap(&arr[j], &arr[j+1]);
}
 

void printArray(int arr[], int size)
{
    int i;
    for (i=0; i < size; i++)
        printf("%d ", arr[i]);
    printf("\n");
}
 

int main()
{
    int nm;
    scanf("%d",&nm);
    int arr[nm];
    for(int i=0;i<nm;i++)
        arr[i]=rand();
    clock_t start_t, end_t;
    start_t = clock();
    bubbleSort(arr, nm);
    end_t= clock();
    printf("\ntime elapsed is %.1f",((double)(end_t - start_t)/CLOCKS_PER_SEC)*1000);
    
    return 0;
}
